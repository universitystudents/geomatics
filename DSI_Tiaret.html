    "PET": os.path.join(data_dir, "DSI_PET_Tiaret_2022.tif"),
    "POP": os.path.join(data_dir, "DSI_POP_Tiaret_2020.tif"),

# === 2. Fonction de statistiques environnementales ===
def compute_stats(path, title, factor=1.0, fill_nan=0.0):
    with rasterio.open(path) as src:
        data = src.read(1).astype("float32")
        res = src.res
        pixel_area_km2 = abs(res[0] * res[1])  # ⚠️ si projection en degrés (WGS84), ce n’est pas km²
        data = np.where(np.isnan(data), fill_nan, data)
        total = np.sum(data * pixel_area_km2 * factor)
        mean = np.mean(data * factor)
        median = np.median(data * factor)
        return mean, median, total

# === 3. Resampling du NDVI sur la taille de PRECIP ===
with rasterio.open(raster_paths["PRECIP"]) as ref_src:
    ref_shape = (ref_src.height, ref_src.width)

        1,
        out_shape=ref_shape,

mean_ndvi = round(np.mean(ndvi_clean), 3)

import os
import numpy as np
import rasterio
from rasterio.enums import Resampling
from rasterio.features import geometry_mask
import geopandas as gpd
import matplotlib.pyplot as plt
import matplotlib.colors as mcolors

# === 1. Dossier de Tiaret
data_dir = r"E:\PFE\DSI\Tiaret"

paths = {
    "ndvi": os.path.join(data_dir, "DSI_NDVI_Tiaret_2022.tif"),
    "precip": os.path.join(data_dir, "DSI_PRECIP_Tiaret_2022.tif"),
    "pet": os.path.join(data_dir, "DSI_PET_Tiaret_2022.tif"),
    "sand": os.path.join(data_dir, "DSI_SAND_Tiaret_0_5cm.tif"),
    "pop": os.path.join(data_dir, "DSI_POP_Tiaret_2020.tif"),
    "shp": os.path.join(data_dir, "Tiaret.shp")
}

# === 2. Lire raster de référence (precip)
with rasterio.open(paths["precip"]) as ref_src:
    ref_data = ref_src.read(1)
    ref_profile = ref_src.profile
    ref_shape = ref_data.shape
    ref_transform = ref_src.transform
    ref_crs = ref_src.crs

# === 3. Fonction pour lire et resampler
def reproject_match(path, ref_profile):
    with rasterio.open(path) as src:
        return src.read(
            out_shape=(1, ref_profile['height'], ref_profile['width']),
            resampling=Resampling.bilinear
        )[0]

# === 4. Lire et aligner tous les rasters
ndvi = reproject_match(paths["ndvi"], ref_profile)
pet = reproject_match(paths["pet"], ref_profile)
sand = reproject_match(paths["sand"], ref_profile)
pop = reproject_match(paths["pop"], ref_profile)

# === 5. Reclassement avec seuils réalistes

# NDVI (végétation)
vqi = np.where(ndvi < 0.15, 2,
       np.where(ndvi < 0.4, 1.5, 1))

# Aridité (P/PET)
ai = ref_data / (pet + 1e-6)
cqi = np.where(ai < 0.15, 2,
        np.where(ai < 0.4, 1.5, 1))

# Sol : faible sable = fragile → SQI = 2
sqi = np.where(sand < 30, 2,
        np.where(sand < 60, 1.5, 1))

# Population (densité)
mqi = np.where(pop > 150, 2,
        np.where(pop > 50, 1.5, 1))

# === 6. Calcul du DSI
dsi = (vqi * cqi * sqi * mqi) ** 0.25

# === 🔴 Fusion des classes moyenne + forte = FORTE
dsi_class = np.where(dsi <= 1.25, 1, 2)

# === 7. Appliquer le masque géographique
gdf = gpd.read_file(paths["shp"]).to_crs(ref_crs)
geoms = [f["geometry"] for f in gdf.__geo_interface__["features"]]
mask = geometry_mask(geoms, transform=ref_transform, invert=True, out_shape=ref_shape)

dsi_class_masked = np.where(mask, dsi_class, np.nan)

# === 8. Statistiques
total_pixels = np.sum(~np.isnan(dsi_class_masked))
class_1 = np.sum(dsi_class_masked == 1)
class_2 = np.sum(dsi_class_masked == 2)

p1 = 100 * class_1 / total_pixels
p2 = 100 * class_2 / total_pixels

print(f"🟢 Faible sensibilité : {p1:.2f}%")
print(f"🔴 Forte sensibilité (moyenne + forte) : {p2:.2f}%")

# === 9. Carte binaire DSI
cmap = mcolors.ListedColormap(['green', 'red'])
bounds = [0.5, 1.5, 2.5]
norm = mcolors.BoundaryNorm(bounds, cmap.N)

plt.figure(figsize=(10, 6))
plt.imshow(dsi_class_masked, cmap=cmap, norm=norm)
cbar = plt.colorbar(ticks=[1, 2])
cbar.ax.set_yticklabels(['Faible', 'Forte'])
cbar.set_label("Classe de sensibilité à la désertification")
plt.title("Carte DSI binaire – Wilaya de Tiaret")
plt.axis("off")
plt.tight_layout()
plt.show()

# === 10. Graphiques simplifiés
labels = ['Faible', 'Forte']
sizes = [p1, p2]
colors = ['green', 'red']

# Camembert
plt.figure(figsize=(7, 7))
plt.pie(sizes, labels=labels, autopct='%1.1f%%', colors=colors, startangle=140)
plt.title("Répartition du DSI – Tiaret (2 classes)")
plt.show()

# Barres
plt.figure(figsize=(7, 5))
plt.bar(labels, sizes, color=colors)
plt.ylabel("Pourcentage de la surface (%)")
plt.title("Surface de chaque classe DSI – Tiaret (2 classes)")
plt.ylim(0, 100)
plt.grid(axis='y', linestyle='--', alpha=0.5)
plt.tight_layout()
plt.show()

# === 11. Export GeoTIFF classé
output_path = os.path.join(data_dir, "DSI_BINAIRE_Tiaret.tif")
profile_out = ref_profile.copy()
profile_out.update(dtype='uint8', nodata=0)

with rasterio.open(output_path, "w", **profile_out) as dst:
    dst.write(np.nan_to_num(dsi_class_masked, nan=0).astype("uint8"), 1)
