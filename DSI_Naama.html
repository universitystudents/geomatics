import os
import numpy as np
import rasterio
import pandas as pd
import matplotlib.pyplot as plt
from rasterio.enums import Resampling

# === 1. Dossier et chemins ===
data_dir = r"E:\PFE\DSI\Naama"

raster_paths = {
    "NDVI": os.path.join(data_dir, "DSI_NDVI_Naama_2022.tif"),
    "PRECIP": os.path.join(data_dir, "DSI_PRECIP_Naama_2022.tif"),
    "PET": os.path.join(data_dir, "DSI_PET_Naama_2022.tif"),
    "POP": os.path.join(data_dir, "DSI_POP_Naama_2020.tif"),
    "SAND": os.path.join(data_dir, "DSI_SAND_Naama_0_5cm.tif"),
}

# === 2. Lire la couche de rfrence (PRECIP) pour dfinir la taille
with rasterio.open(raster_paths["PRECIP"]) as ref_src:
    ref_profile = ref_src.profile
    ref_shape = (ref_profile["height"], ref_profile["width"])

# === 3. Fonction pour resampler et analyser les statistiques
def resample_to_ref(path, ref_shape):
    with rasterio.open(path) as src:
        return src.read(
            1,
            out_shape=ref_shape,
            resampling=Resampling.bilinear
        ).astype("float32")

def compute_stats(path, title, factor=1.0, fill_nan=0.0):
    data = resample_to_ref(path, ref_shape)
    with rasterio.open(path) as src:
        res = src.res
        pixel_area_km2 = abs(res[0] * res[1])
    data = np.where(np.isnan(data), fill_nan, data)
    total = np.sum(data * pixel_area_km2 * factor)
    mean = np.mean(data * factor)
    median = np.median(data * factor)
    return mean, median, total

# === 4. Calculs des indicateurs ===
indicateurs = []

# 1. NDVI
ndvi_data = resample_to_ref(raster_paths["NDVI"], ref_shape)
ndvi_clean = ndvi_data[~np.isnan(ndvi_data)]

# 4. Teneur en sable
sand_mean, sand_median, _ = compute_stats(raster_paths["SAND"], "Sable", factor=0.1)

import os
import numpy as np
import rasterio
from rasterio.enums import Resampling
from rasterio.features import geometry_mask
import geopandas as gpd
import matplotlib.pyplot as plt
import matplotlib.colors as mcolors

# === 1. Dossier et chemins
data_dir = r"E:\PFE\DSI\Naama"

paths = {
    "ndvi": os.path.join(data_dir, "DSI_NDVI_Naama_2022.tif"),
    "precip": os.path.join(data_dir, "DSI_PRECIP_Naama_2022.tif"),
    "pet": os.path.join(data_dir, "DSI_PET_Naama_2022.tif"),
    "sand": os.path.join(data_dir, "DSI_SAND_Naama_0_5cm.tif"),
    "pop": os.path.join(data_dir, "DSI_POP_Naama_2020.tif"),
    "shp": os.path.join(data_dir, "Naama.shp")
}

# === 2. Lire raster de rfrence (PRECIP)
with rasterio.open(paths["precip"]) as ref_src:
    ref_data = ref_src.read(1)
    ref_profile = ref_src.profile
    ref_shape = ref_data.shape
    ref_transform = ref_src.transform
    ref_crs = ref_src.crs

# === 3. Fonction de resample
def reproject_match(path, ref_profile):
    with rasterio.open(path) as src:
        return src.read(
            out_shape=(1, ref_profile['height'], ref_profile['width']),
            resampling=Resampling.bilinear
        )[0]

ndvi = reproject_match(paths["ndvi"], ref_profile)
pet = reproject_match(paths["pet"], ref_profile)
sand_raw = reproject_match(paths["sand"], ref_profile)
pop = reproject_match(paths["pop"], ref_profile)

# === 4. Reclassement (seuils stricts)
vqi = np.where(ndvi < 0.10, 2, np.where(ndvi < 0.35, 1.5, 1))
ai = ref_data / (pet + 1e-6)
cqi = np.where(ai < 0.10, 2, np.where(ai < 0.30, 1.5, 1))
sand = sand_raw / 10
sqi = np.where(sand < 35, 2, np.where(sand < 50, 1.5, 1))
mqi = np.where(pop > 150, 2, np.where(pop > 50, 1.5, 1))

# === 5. Calcul du DSI
dsi = (vqi * cqi * sqi * mqi) ** 0.25

# ===  6. Reclassement binaire
dsi_class = np.where(dsi <= 1.25, 1, 2)

# === 7. Appliquer le masque shapefile
gdf = gpd.read_file(paths["shp"]).to_crs(ref_crs)
geoms = [f["geometry"] for f in gdf.__geo_interface__["features"]]
mask = geometry_mask(geoms, transform=ref_transform, invert=True, out_shape=ref_shape)

dsi_class_masked = np.where(mask, dsi_class, np.nan)

# === 8. Statistiques
total_pixels = np.sum(~np.isnan(dsi_class_masked))
class_1 = np.sum(dsi_class_masked == 1)
class_2 = np.sum(dsi_class_masked == 2)

p1 = 100 * class_1 / total_pixels
p2 = 100 * class_2 / total_pixels

print(f" Rsultats pour Nama")
print(f" Faible sensibilit : {p1:.2f}%")
print(f" Forte sensibilit (moyenne + forte) : {p2:.2f}%")

# === 9. Carte binaire
cmap = mcolors.ListedColormap(['green', 'red'])
bounds = [0.5, 1.5, 2.5]
norm = mcolors.BoundaryNorm(bounds, cmap.N)

plt.figure(figsize=(10, 6))
plt.imshow(dsi_class_masked, cmap=cmap, norm=norm)
cbar = plt.colorbar(ticks=[1, 2])
cbar.ax.set_yticklabels(['Faible', 'Forte'])
cbar.set_label("Classe de sensibilit  la dsertification")
plt.title("Carte DSI binaire  Wilaya de Nama")
plt.axis("off")
plt.tight_layout()
plt.show()

# === 10. Graphiques
labels = ['Faible', 'Forte']
sizes = [p1, p2]
colors = ['green', 'red']

# Camembert
plt.figure(figsize=(7, 7))
plt.pie(sizes, labels=labels, autopct='%1.1f%%', colors=colors, startangle=140)
plt.title("Rpartition du DSI  Nama (2 classes)")
plt.show()

# Barres
plt.figure(figsize=(7, 5))
plt.bar(labels, sizes, color=colors)
plt.ylabel("Pourcentage de la surface (%)")
plt.title("Surface de chaque classe DSI  Nama (2 classes)")
plt.ylim(0, 100)
plt.grid(axis='y', linestyle='--', alpha=0.5)
plt.tight_layout()
plt.show()

# === 11. Export GeoTIFF
output_path = os.path.join(data_dir, "DSI_BINAIRE_Naama.tif")
profile_out = ref_profile.copy()
profile_out.update(dtype='uint8', nodata=0)

with rasterio.open(output_path, "w", **profile_out) as dst:
    dst.write(np.nan_to_num(dsi_class_masked, nan=0).astype("uint8"), 1)
